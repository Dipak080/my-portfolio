How to add Razorpay Payment Gateway in Laravel

Accept online payments in Laravel using Razorpay — install the SDK, create an order, open the checkout, and verify the payment signature on your server.

1. Install the SDK & set your keys

bash · .env
# Install the official Razorpay PHP SDK
composer require razorpay/razorpay

# Add your API keys to .env (get them from the Razorpay dashboard)
RAZORPAY_KEY=rzp_test_xxxxxxxxxxxxx
RAZORPAY_SECRET=your_secret_key_here

2. Add routes

routes/web.php
use App\Http\Controllers\PaymentController;

Route::get('/pay', [PaymentController::class, 'checkout']);
Route::post('/payment/verify', [PaymentController::class, 'verify'])->name('payment.verify');

3. Create the controller

app/Http/Controllers/PaymentController.php
<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use Razorpay\Api\Api;

class PaymentController extends Controller
{
    public function checkout()
    {
        $api = new Api(env('RAZORPAY_KEY'), env('RAZORPAY_SECRET'));

        $order = $api->order->create([
            'receipt'  => 'rcpt_' . uniqid(),
            'amount'   => 500 * 100,       // amount in paise (= ₹500)
            'currency' => 'INR',
        ]);

        return view('checkout', [
            'orderId' => $order['id'],
            'amount'  => $order['amount'],
            'key'     => env('RAZORPAY_KEY'),
        ]);
    }

    public function verify(Request $request)
    {
        $api = new Api(env('RAZORPAY_KEY'), env('RAZORPAY_SECRET'));

        try {
            $api->utility->verifyPaymentSignature([
                'razorpay_order_id'   => $request->razorpay_order_id,
                'razorpay_payment_id' => $request->razorpay_payment_id,
                'razorpay_signature'  => $request->razorpay_signature,
            ]);

            return 'Payment successful and verified!';
        } catch (\Exception $e) {
            return 'Payment verification failed: ' . $e->getMessage();
        }
    }
}

JWT Authentication in CodeIgniter 4

Issue a signed JSON Web Token on login using firebase/php-jwt — the same approach I used to secure the TaskOps API.

app/Controllers/Api/Auth.php
<?php
// composer require firebase/php-jwt

namespace App\Controllers\Api;

use CodeIgniter\RESTful\ResourceController;
use Firebase\JWT\JWT;
use App\Models\UserModel;

class Auth extends ResourceController
{
    public function login()
    {
        $json = $this->request->getJSON();
        $user = (new UserModel())->where('email', $json->email)->first();

        if (! $user || ! password_verify($json->password, $user['password'])) {
            return $this->failUnauthorized('Invalid credentials');
        }

        $payload = [
            'iss' => base_url(),
            'sub' => $user['id'],
            'iat' => time(),
            'exp' => time() + 3600,        // expires in 1 hour
        ];

        $token = JWT::encode($payload, getenv('JWT_SECRET'), 'HS256');

        return $this->respond(['token' => $token, 'user' => $user]);
    }
}
PHP · REST API

Build a REST API in CodeIgniter 4

A full CRUD resource controller with JSON responses and correct HTTP status codes.

app/Controllers/Api/Tasks.php
<?php

namespace App\Controllers\Api;

use CodeIgniter\RESTful\ResourceController;
use App\Models\TaskModel;

class Tasks extends ResourceController
{
    protected $modelName = TaskModel::class;
    protected $format    = 'json';

    // GET /api/tasks
    public function index()
    {
        return $this->respond($this->model->findAll());
    }

    // GET /api/tasks/1
    public function show($id = null)
    {
        $task = $this->model->find($id);
        return $task ? $this->respond($task) : $this->failNotFound('Task not found');
    }

    // POST /api/tasks
    public function create()
    {
        $data = $this->request->getJSON(true);
        $id   = $this->model->insert($data);
        return $this->respondCreated(array_merge(['id' => $id], $data));
    }

    // PUT /api/tasks/1
    public function update($id = null)
    {
        $this->model->update($id, $this->request->getJSON(true));
        return $this->respond(['updated' => true]);
    }

    // DELETE /api/tasks/1
    public function delete($id = null)
    {
        $this->model->delete($id);
        return $this->respondDeleted(['deleted' => true]);
    }
}
JavaScript · React

Data Fetching in React with Hooks

Load data from an API using useState and useEffect, with loading state handled cleanly.

TaskList.jsx
import { useState, useEffect } from 'react';

export default function TaskList() {
  const [tasks, setTasks]     = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('https://api.example.com/tasks')
      .then((res) => res.json())
      .then((data) => {
        setTasks(data);
        setLoading(false);
      })
      .catch(() => setLoading(false));
  }, []);

  if (loading) return <p>Loading…</p>;

  return (
    <ul>
      {tasks.map((task) => (
        <li key={task.id}>{task.title}</li>
      ))}
    </ul>
  );
}
React Native

Render an API List in React Native

Fetch from a REST API and display the results in a performant FlatList with a loading spinner.

Tasks.js
import React, { useEffect, useState } from 'react';
import { View, Text, FlatList, ActivityIndicator } from 'react-native';

export default function Tasks() {
  const [tasks, setTasks]     = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('https://api.example.com/tasks')
      .then((res) => res.json())
      .then((data) => setTasks(data))
      .finally(() => setLoading(false));
  }, []);

  if (loading) return <ActivityIndicator size="large" />;

  return (
    <FlatList
      data={tasks}
      keyExtractor={(item) => item.id.toString()}
      renderItem={({ item }) => (
        <View style={{ padding: 16 }}>
          <Text style={{ fontSize: 16 }}>{item.title}</Text>
        </View>
      )}
    />
  );
}
MySQL · PHP PDO

MySQL CRUD with PHP PDO (prepared statements)

Safe Create / Read / Update / Delete queries using prepared statements to prevent SQL injection.

db.php
<?php
$pdo = new PDO('mysql:host=localhost;dbname=taskops', 'root', '', [
    PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
]);

// CREATE
$stmt = $pdo->prepare('INSERT INTO tasks (title, status) VALUES (?, ?)');
$stmt->execute(['Design homepage', 'open']);

// READ
$tasks = $pdo->query('SELECT * FROM tasks')->fetchAll(PDO::FETCH_ASSOC);

// UPDATE
$stmt = $pdo->prepare('UPDATE tasks SET status = ? WHERE id = ?');
$stmt->execute(['done', 1]);

// DELETE
$stmt = $pdo->prepare('DELETE FROM tasks WHERE id = ?');
$stmt->execute([1]);

MongoDB CRUD with Mongoose

Connect to MongoDB and run CRUD operations with a Mongoose schema in Node.js.

tasks.js
const mongoose = require('mongoose');

await mongoose.connect('mongodb://127.0.0.1:27017/taskops');

const Task = mongoose.model('Task', new mongoose.Schema({
  title:  String,
  status: { type: String, default: 'open' },
}));

// CREATE
const task = await Task.create({ title: 'Write docs' });

// READ
const tasks = await Task.find();

// UPDATE
await Task.findByIdAndUpdate(task._id, { status: 'done' });

// DELETE
await Task.findByIdAndDelete(task._id);
